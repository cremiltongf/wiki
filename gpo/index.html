Papel de parede

Script de papel de parede

Conteúdo do Script

@echo off

SET "IMGIF=\\192.168.X.XX\wallpaper.png"
SET "USERIF=aluno"

if not exist "C:\Users\%USERIF%\Pictures\wallpaper.png" (
    xcopy /y "%IMGIF%" "C:\Users\%USERIF%\Pictures"
)

exit
