Instalações de Programas

Chrome

Silent install

                    
                    sc config "gupdate" start= auto
                    sc config "gupdatem" start= auto
                    sc config "GoogleChromeElevationService" start= auto

                    "ChromeStandaloneSetup64.exe" /silent /install
                    IF %ERRORLEVEL% EQU 0 echo CHROME instalado com sucesso

Atalhos

                    
                    del "C:\Users\aluno\Desktop\Google Chrome.lnk"
                    del "C:\Users\Public\Desktop\Google Chrome.lnk"

Firefox

Silent install

                    
                    "Firefox Setup 97.0.exe" /s
                    IF %ERRORLEVEL% EQU 0 echo FIREFOX instalado com sucesso

Silent uninstall

                    
                    "%ProgramFiles%\Mozilla Firefox\uninstall\helper.exe" /S

Atalhos

                    
                    del "C:\Users\aluno\Desktop\Firefox.lnk"
                    del "C:\Users\Public\Desktop\Firefox.lnk"

Git

Silent install

                    
                    "Git-2.35.1.2-64-bit.exe" /VERYSILENT /NORESTART /COMPONENTS="icons,icons\desktop,ext,ext\shellhere,ext\guihere,gitlfs,assoc,assoc_sh"

Silent uninstall

                    
                    "%ProgramFiles(x86)%\Git\unins000.exe" /VERYSILENT /NORESTART
                    "%ProgramFiles%\Git\unins000.exe" /VERYSILENT /NORESTART
