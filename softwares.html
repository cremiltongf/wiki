Instalações de Programas

Chrome

Silent install

                    
                    sc config "gupdate" start= auto
                    sc config "gupdatem" start= auto
                    sc config "GoogleChromeElevationService" start= auto

                    "ChromeStandaloneSetup64.exe" /silent /install
                    IF %ERRORLEVEL% EQU 0 echo CHROME instalado com sucesso

Atalhos

                    
                    del "C:\Users\aluno\Desktop\Google Chrome.lnk"
                    del "C:\Users\Public\Desktop\Google Chrome.lnk"
